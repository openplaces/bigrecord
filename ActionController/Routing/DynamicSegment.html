def match_extraction(next_capture)
  # All non code-related keys (such as :id, :slug) are URI-unescaped as
  # path parameters.
  default_value = default ? default.inspect : nil
  %[
    value = if (m = match[#{next_capture}])
      CGI.unescape(m)
    else
      #{default_value}
    end
    params[:#{key}] = value if value
  ]
end
